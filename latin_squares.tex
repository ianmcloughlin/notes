\section{Finite Sets}
$S = \{a,b,c\}$ \\[3mm]
$|S| = 3$ \\[3mm]
$[n] = \{0, 1, 2, \ldots, n-1\}$ \\[3mm]
$|[n]| = n$ \\[3mm]


\section{Latin Square}

$L = \{ (i,j,k) \} \subset [n]^3$ \\
such that \\
$\forall x \in [n]: \{(y,k) \; \vert \; (x,y,k) \in L\} = [n]^2$\\
and\\
$\forall y \in [n]: \{(x,k) \; \vert \; (x,y,k) \in L\} = [n]^2.$
